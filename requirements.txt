# --- PyTorch CPU-only --------------------------------------
# sentence-transformers depende de torch. Por defecto, en Linux/Windows
# pip descargaría la build con CUDA (~5 GB de paquetes nvidia-*) que no
# sirve para inferencia en CPU. Este índice extra fuerza la rueda "+cpu"
# (~200 MB). En macOS la rueda oficial de PyPI ya es CPU.
--extra-index-url https://download.pytorch.org/whl/cpu
torch

# --- Framework Web ---
# starlette se fija explícito: versiones previas a 1.3.1 tienen varios
# DoS conocidos (CVE-2025-62727, CVE-2026-54283) y el server escucha en
# 0.0.0.0, o sea que es alcanzable desde la red local.
fastapi==0.139.0
starlette==1.3.1
uvicorn[standard]==0.34.0

# --- Motor de IA / LLM local ---
# Se mantiene la rama 0.3.x de langchain: la 1.x rompe la API del RAG.
# langchain-core >= 0.3.85 corrige la inyección por serialización
# (CVE-2025-68664, crítica) y la inyección de plantillas (CVE-2025-65106).
langchain==0.3.30
langchain-core==0.3.86
langchain-community==0.3.31
langchain-text-splitters==1.1.2
langchain-ollama==0.3.10
langchain-huggingface==0.3.1
langsmith==0.8.18

# --- Base de datos vectorial ---
# 0.6.3 (mismo formato de BD que 0.5.x) levanta el tope tokenizers<=0.20.3
# que impedía instalar transformers >= 4.48.
chromadb==0.6.3

# --- Embeddings ligeros ---
# transformers se fija explícito: < 4.53 arrastra una docena de CVEs
# altos (ReDoS y deserialización insegura en loaders de modelos).
sentence-transformers==3.3.1
transformers==4.57.6

# --- Procesamiento de PDFs ---
# pypdf 5.x acumulaba ~30 CVEs de DoS (loops infinitos / agotamiento de
# RAM con PDFs malformados). Todos corregidos en 6.13.3.
pypdf==6.13.3

# --- Utilidades ---
python-multipart==0.0.31
pydantic==2.10.4
